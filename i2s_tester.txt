#include <Arduino.h>
#include <driver/i2s.h>
#include <math.h>

// I2S Pins
#define I2S_BCLK 5
#define I2S_LRC  6
#define I2S_DOUT 7

// Audio properties
#define SAMPLE_RATE 44100
#define SINE_FREQ 440.0 // 440 Hz (A4 note)
#define AMPLITUDE 10000 // Max is 32767 for 16-bit

void setup() {
    Serial.begin(115200);
    delay(1000);
    Serial.println("\n\n=== Pure ESP-IDF I2S Tone Tester ===");

    i2s_config_t i2s_config = {
        .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_TX),
        .sample_rate = SAMPLE_RATE,
        .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,
        .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT,
        .communication_format = I2S_COMM_FORMAT_STAND_I2S,
        .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
        .dma_buf_count = 8,
        .dma_buf_len = 64,
        .use_apll = false,
        .tx_desc_auto_clear = true,
        .fixed_mclk = 0
    };

    i2s_pin_config_t pin_config = {
        .bck_io_num = I2S_BCLK,
        .ws_io_num = I2S_LRC,
        .data_out_num = I2S_DOUT,
        .data_in_num = I2S_PIN_NO_CHANGE
    };

    Serial.println("Installing I2S driver...");
    if (i2s_driver_install(I2S_NUM_0, &i2s_config, 0, NULL) != ESP_OK) {
        Serial.println("FATAL: Failed to install I2S driver!");
        while (true) delay(10);
    }

    Serial.println("Setting I2S pins...");
    if (i2s_set_pin(I2S_NUM_0, &pin_config) != ESP_OK) {
        Serial.println("FATAL: Failed to set I2S pins!");
        while (true) delay(10);
    }

    Serial.println("I2S driver success. Pumping a 440Hz sine wave...");
}

void loop() {
    // Generate a chunk of sine wave
    size_t bytes_written;
    int16_t sample_buffer[64 * 2]; // 64 samples * 2 channels (Left and Right)
    static float phase = 0.0;
    
    for (int i = 0; i < 64; i++) {
        int16_t sample = (int16_t)(AMPLITUDE * sin(phase));
        
        sample_buffer[i * 2] = sample;       // Left channel
        sample_buffer[(i * 2) + 1] = sample; // Right channel
        
        phase += 2.0 * PI * SINE_FREQ / SAMPLE_RATE;
        if (phase >= 2.0 * PI) {
            phase -= 2.0 * PI;
        }
    }
    
    // Write audio buffer directly to I2S DMA
    i2s_write(I2S_NUM_0, sample_buffer, sizeof(sample_buffer), &bytes_written, portMAX_DELAY);
}
